      END FOR

    END IF
  END MAIN LOOP

  ON EVENT change of [Switch 1]:
    Restart main loop
  END EVENT

Example 15:
DEFINE PROCEDURE signal(color):
  Set color of [Lamp 3] to value of 'color'
  Wait 100 milliseconds
  Execute turnOff on [Lamp 3]
  Wait 100 milliseconds
  Set color of [Lamp 3] to value of 'color'
  Wait 100 milliseconds
  Execute turnOff on [Lamp 3]
  Wait 500 milliseconds
END PROCEDURE

START
  ON EVENT change of [Touch Button 1]:
    WHILE (get property 'touched' from [Touch Button 1]) is TRUE:
      CALL procedure signal(color: #33ff33)
    END WHILE
  END EVENT

  ON EVENT change of [Touch Button 2]:
    WHILE (get property 'touched' from [Touch Button 2]) is TRUE:
      CALL procedure signal(color: #ff0000)
    END WHILE
  END EVENT
END

Example 24:
START
  // on_start block
  Set variable 'estado' = FALSE
  Execute clear on [LED Matrix 1]

  // Event handler
  ON EVENT suddenIncrease of [Noise Level 1]:
    Set variable 'estado' = NOT (get variable 'estado')
    IF (get variable 'estado') is TRUE THEN
      // Animate ON
      FOR 'lineas' FROM 1 TO 7 STEP 1:
        FOR 'columnas' FROM 1 TO 7 STEP 1:
          Draw pixel at ('lineas', 'columnas') with color #ff4444 on [LED Matrix 1]
          Wait 10 milliseconds
        END FOR
      END FOR
    ELSE
      // Animate OFF
      FOR 'lineas' FROM 7 TO 1 STEP -1:
        FOR 'columnas' FROM 7 TO 1 STEP -1:
          Draw pixel at ('lineas', 'columnas') with color #000000 on [LED Matrix 1]
          Wait 10 milliseconds
        END FOR
      END FOR
    END IF
  END EVENT
END

Example 25:
START
  // on_start block
  Set variable 'sonidoFijo' = TRUE

  // Main loop
  MAIN LOOP:
    Set background of [Led Matrix 2] to #cccccc
    Let posX = 4 + (-1 * (get property 'oriX' from [Inclination 1]))
    Let posY = 4 + (get property 'oriY' from [Inclination 1])

    IF (posX > -2 AND posX < 2) AND (posY > -2 AND posY < 2) THEN
      // In target
      Draw pixel on [Led Matrix 2] at (posX, posY) with color #33cc00
      IF (get variable 'sonidoFijo') is TRUE THEN
        Execute play on [Musical Keyboard 3] with note value 60
        Set variable 'sonidoFijo' = FALSE
      END IF
    ELSE
      // Out of target
      Let constrainedX = constrain(posX, between 1 and 7)
      Let constrainedY = constrain(posY, between 1 and 7)
      Draw pixel at (constrainedX, constrainedY) with color #ff4444
      Set variable 'sonidoFijo' = TRUE
      Execute stop on [Musical Keyboard 3]
      Wait 100 milliseconds
      Execute play on [Musical Keyboard 3] with note value 79
    END IF
  END MAIN LOOP
END
